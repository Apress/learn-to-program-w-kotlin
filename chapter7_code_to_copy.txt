package pfb.basics

fun main(args: Array<String>) {
    val str = "Mind the   gap!"
    var compacted = ""
    var previousCharWasASpace = false
    for (c in str) {
        if (c == ' ') {
            if (!previousCharWasASpace) {
                compacted = compacted + c
            }
            previousCharWasASpace = true
        } else {
            compacted = compacted + c
            previousCharWasASpace = false
        }
    }
    println(compacted)
}
