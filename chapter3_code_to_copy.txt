fun tileColors(): Array<Array<Int>> {
    val shades = Array<Array<Int>>(5) {
        Array<Int>(5) { 0 }
    }
    //Set each cell in row 0 to be black
    for (col in 0..4) {
        shades[0][col] = 0
    }
    //Row 1 is dark grey.
    for (col in 0..4) {
        shades[1][col] = 65
    }
    //Row 2 is grey.
    for (col in 0..4) {
        shades[2][col] = 130
    }
    //Row 3 is light grey.
    for (col in 0..4) {
        shades[3][col] = 195
    }
    //Row 4 is white.
    for (col in 0..4) {
        shades[4][col] = 255
    }
    return shades
}
